PW Sales = 
var WeekNumber = SELECTEDVALUE(DimDate[Week_Of_Year])
var WeekYear = SELECTEDVALUE(DimDate[Year])
var MaxWeekNumber = CALCULATE(MAX(DimDate[Week_Of_Year]),ALL(DimDate))
RETURN
SUMX(
    FILTER(
        ALL(DimDate),
        IF(WeekNumber=1
        ,DimDate[Week_Of_Year] = MaxWeekNumber && 
        DimDate[Year] = WeekYear-1
        ,DimDate[Week_Of_Year] = WeekNumber - 1 && DimDate[Year] = WeekYear)
    ),
    [Sales])
